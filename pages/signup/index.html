-- Create users table CREATE TABLE users ( id BIGSERIAL PRIMARY KEY, email VARCHAR(255) UNIQUE NOT NULL, name VARCHAR(255) NOT NULL, password VARCHAR(255) NOT NULL, created_at TIMESTAMP DEFAULT NOW(), updated_at TIMESTAMP DEFAULT NOW() ); -- Enable Row Level Security ALTER TABLE users ENABLE ROW LEVEL SECURITY; -- Drop existing policies if they exist DROP POLICY IF EXISTS "Allow insert for anonymous users" ON users; DROP POLICY IF EXISTS "Allow select for anonymous users" ON users; -- Create new policies CREATE POLICY "Allow insert for anonymous users" ON users FOR INSERT WITH CHECK (true); CREATE POLICY "Allow select for anonymous users" ON users FOR SELECT USING (true); CREATE POLICY "Allow update for authenticated users" ON users FOR UPDATE USING (true) WITH CHECK (true);
